Render System

IRenderBackend

The central RHI interface implemented by both OpenGL and Vulkan backends.

class IRenderBackend {
public:
    virtual bool init() = 0;
    virtual void shutdown() = 0;
    virtual void begin_frame() = 0;
    virtual void end_frame() = 0;
    virtual void set_viewport(int x, int y, int w, int h) = 0;
    virtual void set_depth_test(bool enabled) = 0;
    virtual void set_blend(bool enabled, BlendMode mode) = 0;
    virtual void clear(float r, float g, float b, float a) = 0;
    virtual MeshHandle create_mesh(const VertexLayout& layout) = 0;
    virtual ShaderHandle create_shader(const std::string& vs, const std::string& fs) = 0;
    virtual TextureHandle create_texture(int w, int h, const void* data) = 0;
    virtual FramebufferHandle create_framebuffer(int w, int h) = 0;
    virtual void draw_mesh(MeshHandle mesh, ShaderHandle shader) = 0;
    virtual void draw_indexed(MeshHandle mesh, ShaderHandle shader, uint32_t index_count) = 0;
};

GLBackend

OpenGL 4.6 backend implementation. Performs glewInit on startup, logs GPU vendor info, and supports WGL_NV_delay_before_swap for frame pacing.

GLShader

Wraps vertex and fragment shader compilation, uniform uploads, and program linking.

class GLShader {
public:
    bool compile(const std::string& vertex_src, const std::string& fragment_src);
    void set_int(const std::string& name, int value);
    void set_float(const std::string& name, float value);
    void set_vec2(const std::string& name, float x, float y);
    void set_vec3(const std::string& name, float x, float y, float z);
    void set_vec4(const std::string& name, float x, float y, float z, float w);
    void set_mat4(const std::string& name, const float* m);
    int get_uniform_location(const std::string& name); // returns -1 if missing
};

GLMesh

Manages VAO, VBO, and EBO. Uploads vertex data and draws via glDrawArrays or indexed drawing.

class GLMesh {
public:
    void upload_vertices(const void* data, size_t size, const VertexLayout& layout);
    void set_layout(const VertexLayout& layout);
    void draw();
    void draw_indexed(uint32_t count);
};

GLRenderer2D

CPU-side batching renderer for 2D primitives and text.

class GLRenderer2D {
public:
    void draw_rect(const Vector2f& pos, const Vector2f& size, const Vector4f& color);
    void draw_circle(const Vector2f& center, float radius, const Vector4f& color);
    void draw_polygon(const std::vector<Vector2f>& points, const Vector4f& color);
    void draw_text(const std::string& text, const Vector2f& pos, float scale, const Vector4f& color);
    void flush(); // submits batched geometry
};

RenderCommandBuffer

Double-buffered command queue with mutex and condition variable. Main thread submit()s commands; render thread acquire()s them.

class RenderCommandBuffer {
public:
    void submit(RenderCommand cmd);
    std::vector<RenderCommand> acquire();
};

RenderThread

Dedicated background thread that loops, waiting for commands and executing them on the active backend.

class RenderThread {
public:
    void start(IRenderBackend* backend);
    void stop();
    void thread_loop(); // waits on condition_variable, executes acquired commands
};

VulkanBackend

Vulkan 1.2 backend managing instance, surface, logical device, and swapchain.

VulkanDevice

Selects a physical device with graphics and present queues. Uses VMA for memory management.

class VulkanDevice {
public:
    bool create(VkInstance instance, VkSurfaceKHR surface);
    VkPhysicalDevice physical_device() const;
    VkDevice device() const;
    uint32_t graphics_queue_family() const;
    uint32_t present_queue_family() const;
};

VulkanSwapchain

Manages VkSwapchainKHR lifecycle, image views, and framebuffers.

class VulkanSwapchain {
public:
    bool create(VulkanDevice* device, int width, int height);
    void recreate(int width, int height);
    VkSwapchainKHR handle() const;
    uint32_t image_count() const;
};

VulkanBuffer / VulkanTexture

VMA-allocated GPU resources.

class VulkanBuffer {
public:
    bool create(VulkanDevice* device, size_t size, VkBufferUsageFlags usage, VmaMemoryUsage mem_usage);
    void upload(const void* data, size_t size);
    VkBuffer handle() const;
};

class VulkanTexture {
public:
    bool create(VulkanDevice* device, int w, int h, const void* data);
    VkImageView view() const;
    VkSampler sampler() const;
};

VulkanShader

SPIR-V modules with descriptor set layouts and pipeline layouts.

class VulkanShader {
public:
    bool load_spirv(const std::string& vert_path, const std::string& frag_path);
    void bind(VkCommandBuffer cmd);
    void set_push_constant(VkCommandBuffer cmd, const std::string& name, const void* data);
    VkPipelineLayout pipeline_layout() const;
};

Gryce Engine

Gryce Engine is a C++23 game engine built from scratch. It features dual rendering backends (OpenGL 4.6 and Vulkan 1.2), a custom ECS, multi-threaded command-buffer rendering, and pluggable physics/audio modules.
