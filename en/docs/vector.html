Vector

Vector2f / Vector3f / Vector4f

Handwritten vector types with full operator overloading. No external dependency (glm is optional).

#include <GryceRC/GryceMath.h>

struct Vector2f {
    float x, y;
    Vector2f() = default;
    Vector2f(float x, float y);

    float length() const;
    float length_sq() const;
    Vector2f normalized() const;
    void normalize();
    float dot(const Vector2f& other) const;
    float cross(const Vector2f& other) const;

    Vector2f operator+(const Vector2f& other) const;
    Vector2f operator-(const Vector2f& other) const;
    Vector2f operator*(float scalar) const;
    Vector2f operator/(float scalar) const;
    Vector2f& operator+=(const Vector2f& other);
    Vector2f& operator-=(const Vector2f& other);
    Vector2f& operator*=(float scalar);
    Vector2f& operator/=(float scalar);
    bool operator==(const Vector2f& other) const;
    bool operator!=(const Vector2f& other) const;
    float operator[](int index) const;
    float& operator[](int index);
};

// Vector3f adds cross product, reflect, lerp, slerp
struct Vector3f {
    float x, y, z;
    // ... same interface as Vector2f plus:
    Vector3f cross(const Vector3f& other) const;
    Vector3f reflect(const Vector3f& normal) const;
    Vector3f lerp(const Vector3f& target, float t) const;
    Vector3f slerp(const Vector3f& target, float t) const;
};

// Vector4f adds w component, homogenous division
struct Vector4f {
    float x, y, z, w;
    Vector3f xyz() const;
    Vector3f to_vector3() const; // divide by w
};

设计原理

Vector2f、Vector3f 和 Vector4f 是 gryce-engine 的自定义向量类型，提供完整的运算符重载和常用几何操作。它们不使用 SIMD 指令，保持代码简单和可移植。Vector3f 的 cross product 使用右手定则。所有向量类型支持隐式转换到 float*，可以直接传递给 OpenGL 的 glUniform*fv 和 Vulkan 的 push constant。

数学约定

所有角度操作（如 rotate_x, rotate_y, rotate_z）使用度数而非弧度。这与其他图形库（如 glm 使用弧度）不同，是为了与常用 3D 编辑工具（如 Blender）保持一致。长度和距离计算使用标准的欧几里得范数。normalize 在长度为 0 时返回零向量，不抛出异常。

注意事项

  • Vector 类型不支持 SIMD，在大量计算时可能不如 glm 快。
  • cross 在右手坐标系中使用右手定则。
  • lerp 使用线性插值，slerp 使用球面插值，要求输入向量已归一化。
  • Vector4f 的 to_vector3 执行齐次除法（除以 w），w 为 0 时返回零向量。
