Matrix4f

Column-major storage: m[col * 4 + row].

class Matrix4f {
public:
    static Matrix4f identity();
    static Matrix4f from_diagonal(float x, float y, float z, float w);
    static Matrix4f ortho(float l, float r, float b, float t, float n, float f);
    static Matrix4f perspective(float fov, float aspect, float n, float f);
    static Matrix4f look_at(const Vector3f& eye, const Vector3f& center, const Vector3f& up);
    static Matrix4f rotate(float angle, const Vector3f& axis);
    static Matrix4f scale(float x, float y, float z);
    static Matrix4f translate(float x, float y, float z);

    Matrix4f operator*(const Matrix4f& other) const;
    Vector4f operator*(const Vector4f& v) const;
    float* data();
    const float* data() const;
};

Usage Example

Matrix4f proj = Matrix4f::perspective(60.0f * deg2rad, 16.0f/9.0f, 0.1f, 1000.0f);
Matrix4f view = Matrix4f::look_at(eye, center, up);
Matrix4f model = Matrix4f::translate(pos)
                 * Matrix4f::rotate(angle, axis)
                 * Matrix4f::scale(sx, sy, sz);

Vector3f a{1.0f, 2.0f, 3.0f};
Vector3f b{4.0f, 5.0f, 6.0f};
float dot = a.dot(b);
Vector3f cross = a.cross(b);
Vector3f normalized = a.normalized();

Gryce Engine

Gryce Engine is a C++23 game engine built from scratch. It features dual rendering backends (OpenGL 4.6 and Vulkan 1.2), a custom ECS, multi-threaded command-buffer rendering, and pluggable physics/audio modules.
