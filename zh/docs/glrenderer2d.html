GLRenderer2D

GLRenderer2D

CPU 端批处理的 2D 图元和文本渲染器。自动合并绘制调用，在状态变化或显式 flush() 时提交。

#include <GryceRC/GryceGLRenderer2D.h>

class GLRenderer2D {
public:
    void init(IRenderBackend* backend);
    void shutdown();

    void begin_scene(const Matrix4f& projection);
    void end_scene();

    void draw_rect(const Vector2f& pos, const Vector2f& size, const Vector4f& color);
    void draw_rect(const Vector2f& pos, const Vector2f& size, TextureHandle tex, const Vector4f& tint = Vector4f(1,1,1,1));
    void draw_circle(const Vector2f& center, float radius, const Vector4f& color, int segments = 32);
    void draw_line(const Vector2f& a, const Vector2f& b, const Vector4f& color, float thickness = 1.0f);
    void draw_polygon(const std::vector<Vector2f>& points, const Vector4f& color);
    void draw_text(const std::string& text, const Vector2f& pos, float scale, const Vector4f& color);

    void flush(); // 提交批处理几何体

    uint32_t draw_call_count() const;
    uint32_t vertex_count() const;
};

示例

GLRenderer2D renderer;
renderer.init(backend);

Matrix4f ortho = Matrix4f::ortho(0.0f, 1280.0f, 720.0f, 0.0f, -1.0f, 1.0f);
renderer.begin_scene(ortho);
renderer.draw_rect({100, 100}, {200, 50}, {1.0f, 0.2f, 0.2f, 1.0f});
renderer.draw_text("Hello", {120, 115}, 16.0f, {1,1,1,1});
renderer.end_scene();
