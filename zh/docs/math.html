数学库

Vector2f / Vector3f / Vector4f

引擎自定义的 SIMD-friendly 向量类型，采用结构体直接布局，无虚函数，可直接作为 uniform 数据上传 GPU。所有运算均提供 constexpr 构造函数与内联运算符重载。

Vector3f API

struct Vector3f {
    float x, y, z;

    constexpr Vector3f() : x(0), y(0), z(0) {}
    constexpr Vector3f(float x_, float y_, float z_) : x(x_), y(y_), z(z_) {}

    float dot(const Vector3f& other) const;
    Vector3f cross(const Vector3f& other) const;  // 仅 Vector3f
    float length() const;
    float length_sq() const;
    Vector3f normalized() const;
    void normalize();

    // 运算符重载
    Vector3f operator+(const Vector3f& rhs) const;
    Vector3f operator-(const Vector3f& rhs) const;
    Vector3f operator*(float s) const;
    Vector3f operator/(float s) const;
    Vector3f& operator+=(const Vector3f& rhs);
    Vector3f& operator-=(const Vector3f& rhs);
};

// 使用示例
Vector3f a{1.0f, 2.0f, 3.0f};
Vector3f b{4.0f, 5.0f, 6.0f};
float d = a.dot(b);                 // 32.0
Vector3f c = a.cross(b);            // {-3, 6, -3}
Vector3f n = a.normalized();        // 单位向量

Vector2f / Vector4f 差异

  • Vector2f：含 dot、length、normalized，无 cross。
  • Vector4f：含 dot、length、normalized，无 cross。支持 w 分量（常用于齐次坐标）。

Matrix4f

列主序（Column-Major）4×4 矩阵，底层存储为 float m[16]，索引公式为 m[col * 4 + row]。与 OpenGL 默认矩阵布局一致，可直接作为 glUniformMatrix4fv 或 Vulkan push constant 的数据源。

class Matrix4f {
public:
    float m[16];

    static Matrix4f identity();
    static Matrix4f from_diagonal(float x, float y, float z, float w);

    // 投影矩阵
    static Matrix4f ortho(float left, float right, float bottom, float top,
                          float near_plane, float far_plane);
    static Matrix4f perspective(float fov_rad, float aspect,
                                float near_plane, float far_plane);

    // 视图矩阵
    static Matrix4f look_at(const Vector3f& eye, const Vector3f& center,
                            const Vector3f& up);

    // 变换矩阵
    static Matrix4f translate(float x, float y, float z);
    static Matrix4f translate(const Vector3f& t);
    static Matrix4f rotate(float angle_rad, const Vector3f& axis);
    static Matrix4f scale(float x, float y, float z);
    static Matrix4f scale(const Vector3f& s);

    // 矩阵乘法
    Matrix4f operator*(const Matrix4f& rhs) const;
    Vector4f operator*(const Vector4f& rhs) const;

    // 辅助
    Matrix4f transposed() const;
    Matrix4f inverted() const;
    float* data() { return m; }
    const float* data() const { return m; }
};

完整变换示例

// 投影 * 视图 * 模型
Matrix4f proj = Matrix4f::perspective(60.0f * deg2rad, 16.0f/9.0f, 0.1f, 1000.0f);
Matrix4f view = Matrix4f::look_at(eye, center, up);
Matrix4f model = Matrix4f::translate(pos)
                 * Matrix4f::rotate(angle, axis)
                 * Matrix4f::scale(sx, sy, sz);

Matrix4f mvp = proj * view * model;
shader.set_mat4("u_mvp", mvp);

Gryce Engine

Gryce Engine 是一款从零手写的 C++23 游戏引擎。支持 OpenGL 4.6 与 Vulkan 1.2 双后端渲染，自定义 ECS，多线程命令缓冲区渲染，以及可插拔物理/音频模块。
